\subject{边框参数}

\starttabulate[|l|l|l|]
\HL
\NC \bf 参数 \NC \bf 说明 \NC \bf 默认值 \NC\NR
\HL
\NC frame \NC 边框类型 \NC outer \NC\NR
\NC \NC \bf 可选值: outer, tian, outertian, mi, outermi, x, outerx, none \NC \NC\NR
\HL
\NC framestroke \NC 边框线条颜色 \NC black \NC\NR
\NC framefill \NC 边框填充颜色 \NC (无) \NC\NR
\NC framelinewidth \NC 边框线条宽度 \NC 0.5pt \NC\NR
\NC framedashpattern \NC 边框虚线样式 \NC solid \NC\NR
\NC \NC \bf 可选值: solid, dashed, dotted \NC \NC\NR
\HL
\NC iframestroke \NC 内部边框线条颜色 \NC 0.5white \NC\NR
\NC iframelinewidth \NC 内部边框线条宽度 \NC 0.3pt \NC\NR
\NC iframedashpattern \NC 内部边框虚线样式 \NC dashed \NC\NR
\HL
\stoptabulate
